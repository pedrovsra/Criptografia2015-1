def strxor(a, b):
    return "".join([chr(ord(x) ^ ord(y)) for (x, y) in zip(a, b)])
    
def my_range(start, end, step):
    while start <= end:
        yield start
        start += step

def dehex(a):
    aux = ""
    for x in my_range(0, len(a)-1, 2):
        c = a[x] + a[x+1]
        d = c.decode('hex')
        aux = aux + d
    
    return aux
    
def enc(a):
    return a.encode('hex')
    
def jay(a, b):
    return strxor(dehex(a), dehex(b))
